快速驗證命令
================================================================================

# 查看容器狀態
sudo docker-compose ps

# 查看 micro-ROS Agent 日誌
sudo docker logs micro_ros_agent

# 查看 ROS 2 Core 日誌
sudo docker logs ros2_core

# 驗證 ROS 2 話題
sudo docker exec ros2_core bash -c "source /opt/ros/humble/setup.bash && ros2 topic list"

# 停止所有容器
sudo docker-compose down

# 重新啟動容器
cd ~/ros2_deployment && sudo docker-compose up -d
